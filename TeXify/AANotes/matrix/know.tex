\subsection{知识引导}

\ti{常见分块矩阵的逆矩阵. }
$$
\text { 设 } T=\left(\begin{array}{cc}
    A & B \\
    0 & D
    \end{array}\right) \text {, 其中 } A, D \text { 都是可逆方阵, }  T^{-1}=\left(\begin{array}{cc}
        A^{-1} & -A^{-1} B D^{-1} \\
        0 & D^{-1}
        \end{array}\right)\text {. }
$$

\ti{三种初等矩阵. } 
(1) 第一类初等矩阵
$$
P(i,j)=\begin{bmatrix}
    1 &  &  &  &  &  &  &  & \\
     & \ddots  &  &  &  &  &  &  & \\
     &  &  & 1 &  &  &  &  & \\
     &  &  &  & 0 & \cdots & 1 &  & \\
     &  &  &  & \vdots &  & \vdots &  & \\
     &  &  &  & 1 & \cdots & 0&  & \\
     &  &  &  &  &  &  & 1 & \\
     &  &  &  &  &  &  &  &\ddots \\
     &  &  &  &  &  &  &  & &1
   \end{bmatrix}
$$

(2) 第二类初等矩阵. 
$$
P(i(c))=\begin{bmatrix}
    1 &  &  &  &  &  &  &  & \\
     & \ddots  &  &  &  &  &  &  & \\
     &  &  & 1 &  &  &  &  & \\
     &  &  &  & 1 &  &  &  & \\
     &  &  &  &  & \ddots &  &  & \\
     &  &  &  &  & & c&  & \\
     &  &  &  &  &  &  & 1 & \\
     &  &  &  &  &  &  &  &\ddots \\
     &  &  &  &  &  &  &  & &1
   \end{bmatrix}
$$

(3) 第三类初等矩阵.
$$
P(i,j(c))=\begin{bmatrix}
    1 &  &  &  &  &  &  &  & \\
     & \ddots  &  &  &  &  &  &  & \\
     &  &  & 1 &  &  &  & c & \\
     &  &  &  & 1 &  &  &  & \\
     &  &  &  &  & \ddots &  &  & \\
     &  &  &  &  & & 1&  & \\
     &  &  &  &  &  &  & 1 & \\
     &  &  &  &  &  &  &  &\ddots \\
     &  &  &  &  &  &  &  & &1
   \end{bmatrix}
$$
